JavaScript Cheat Sheet

My notes for important JavaScript concepts and syntax.

Use let and const instead of var. 

let age = 25;
const name = "Kevin";

Basic function syntax:

function add(a, b) {
  return a + b;
}
const multiply = (a, b) => a * b;

JavaScript has dynamic typing.

let x = 10;        // number
let y = "hello";  // string
let z = true;     // boolean

Arrays store multiple values.

let numbers = [1, 2, 3, 4];

Objects store key-value pairs.

let person = {
  name: "Kevin",
  age: 20
};
if (age >= 18) {
  console.log("Adult");
} else {
  console.log("Minor");
}
switch(day) {
  case 1:
    console.log("Monday");
    break;
  default:
    console.log("Other day");
}
for (let i = 0; i < 5; i++) {
  console.log(i);
}
let i = 0;
while (i < 3) {
  console.log(i);
  i++;
}
numbers.forEach(num => {
  console.log(num);
});
let doubled = numbers.map(n => n * 2);
let evens = numbers.filter(n => n % 2 === 0);
